pushFront(key)
  inserAfter(self.head,key)# 노드`key`를 헤드노드 앞에 생성한다

## 탐색연산 
def search(self,key
  v= self.head # dummy node
  while v.next != self.head: # v가 리스트 한바퀴를 돌떄 까지 반복한다
    if v.key == key:
       return key
    v= v.next
  return None
